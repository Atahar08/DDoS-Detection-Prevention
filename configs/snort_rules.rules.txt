# Snort rules for DDoS detection

# SYN flood detection
alert tcp any any -> any 80 (flags: S; msg:"Possible SYN Flood"; sid:1000001; rev:1;)

# HTTP GET flood detection
alert tcp any any -> any 80 (msg:"HTTP GET Flood"; flow:to_server,established; content:"GET"; threshold:type threshold, track by_src, count 50, seconds 10; sid:1000002; rev:1;)
